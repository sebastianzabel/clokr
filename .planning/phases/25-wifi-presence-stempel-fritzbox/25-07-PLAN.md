```typescript
// ── Employee self-service: /me/wifi ────────────────────────────────────────

export interface MyWifiDevice {
  id: string;
  mac: string;
  label: string | null;
}

export interface MyWifiData {
  wifiPresenceEnabled: boolean;
  devices: MyWifiDevice[];
}

export async function getMyWifi(): Promise<MyWifiData> {
  return api.get<MyWifiData>("/me/wifi");
}

export async function updateMyWifi(wifiPresenceEnabled: boolean): Promise<{ wifiPresenceEnabled: boolean }> {
  return api.patch<{ wifiPresenceEnabled: boolean }>("/me/wifi", { wifiPresenceEnabled });
}

export async function addMyDevice(mac: string, label?: string): Promise<MyWifiDevice> {
  return api.post<MyWifiDevice>("/me/wifi/devices", { mac, label: label ?? null });
}

export async function removeMyDevice(id: string): Promise<void> {
  await api.delete(`/me/wifi/devices/${id}`);
}
```

   ```typescript
   async function toggleWifi() {
     try {
       const res = await updateMyWifi(wifiEnabled);
       wifiEnabled = res.wifiPresenceEnabled;
       toasts.success(wifiEnabled ? "WiFi-Präsenzerkennung aktiviert." : "WiFi-Präsenzerkennung deaktiviert.");
     } catch {
       // revert optimistic toggle
       wifiEnabled = !wifiEnabled;
       toasts.error("Fehler beim Speichern.");
     }
   }

   async function addMacDevice() {
     macError = "";
     const mac = newMac.toLowerCase().trim();
     if (!MAC_REGEX.test(mac)) {
       macError = "Ungültige MAC-Adresse. Format: aa:bb:cc:dd:ee:ff";
       return;
     }
     macAdding = true;
     try {
       const device = await addMyDevice(mac, macLabel.trim() || undefined);
       wifiDevices = [...wifiDevices, device];
       newMac = "";
       macLabel = "";
       toasts.success("Gerät wurde hinzugefügt.");
     } catch (e: unknown) {
       macError = e instanceof Error ? e.message : "Gerät konnte nicht gespeichert werden. Bitte versuchen Sie es erneut.";
     } finally {
       macAdding = false;
     }
   }

   async function removeMacDevice(id: string) {
     try {
       await removeMyDevice(id);
       wifiDevices = wifiDevices.filter((d) => d.id !== id);
       toasts.success("Gerät wurde entfernt.");
     } catch {
       toasts.error("Gerät konnte nicht entfernt werden.");
     }
   }
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → SvelteKit proxy → API | Employee submits MAC address input; client-side regex is UX-only — server must also validate |
| employee session → /me/wifi | Only the authenticated employee may read/write their own wifi data; endpoint must enforce identity from JWT, not from request body |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-25-07-01 | Tampering | PATCH /me/wifi (25-05) | mitigate | Server endpoint uses `req.user.employeeId` from JWT — employee cannot spoof another employee's opt-in. Client sends no employeeId in body. Validated in 25-05. |
| T-25-07-02 | Tampering | POST /me/wifi/devices (25-05) | mitigate | MAC format validated server-side with the same regex (25-05); client regex is defense-in-depth only. Duplicate MAC returns 409. |
| T-25-07-03 | Information Disclosure | GET /me/wifi returns device list | accept | Only the authenticated employee receives their own MACs. MACs of other employees are not exposed. Low sensitivity: MAC without context is not strongly identifying. |
| T-25-07-04 | Elevation of Privilege | Regular employee accessing /admin/wifi-presence | accept | Admin route requires ADMIN role (enforced in 25-06 by requireRole). Settings page adds no admin routes. |
| T-25-07-05 | Denial of Service | Employee adds unlimited MACs | mitigate | Server enforces a max-device limit (25-05 should cap at e.g. 10 per employee). If not already implemented, note in summary as a follow-up. |
</threat_model>
