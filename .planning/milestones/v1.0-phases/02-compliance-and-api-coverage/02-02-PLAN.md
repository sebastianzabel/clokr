        it("tenantA admin cannot POST time entry for tenantB employee") -> 403 or 404
        it("tenantA admin cannot DELETE tenantB time entry") -> 403 or 404
      }

      describe("LeaveRequest resource") {
        it("tenantA admin cannot GET tenantB leave requests") -> 403 or 404
        it("tenantA admin cannot POST leave for tenantB employee") -> 403 or 404
      }

      describe("Absence resource") {
        it("tenantA admin cannot GET tenantB absences") -> 403 or 404
      }

      describe("OvertimeAccount resource") {
        it("tenantA admin cannot GET tenantB overtime") -> 403 or 404

      describe("AuditLog resource") {

    ```

      let app, data;

      afterAll: cleanupTestData, closeTestApp()

      // Per Pitfall 7: capture timestamp before each test to isolate audit logs

      describe("TimeEntry mutations") {

        it("DELETE /api/v1/time-entries/:id writes AuditLog with action DELETE")
      }

      describe("Employee mutations") {
        it("POST /api/v1/employees writes AuditLog with action CREATE")
        it("PUT /api/v1/employees/:id writes AuditLog with action UPDATE")
        it("DELETE /api/v1/employees/:id writes AuditLog with action DELETE")
      }

      describe("LeaveRequest mutations") {
        it("POST /api/v1/leave writes AuditLog with action CREATE")
        it("PUT /api/v1/leave/:id (approve/reject) writes AuditLog with action UPDATE")
      }

      describe("Auth mutations") {
        it("POST /api/v1/auth/login (successful) writes AuditLog")
      }

      describe("Settings mutations") {
        it("PUT /api/v1/settings writes AuditLog with action UPDATE")
      }
    }
    ```

    For each test:
    1. Capture `const beforeTs = new Date()` before the mutation
    2. Perform the mutation via app.inject() with proper auth token
    3. Query `app.prisma.auditLog.findMany({ where: { entity: "EntityName", createdAt: { gte: beforeTs } } })`
    4. Assert: at least 1 log entry found
    5. Assert: log entry has non-null userId (should equal data.adminUser.id for admin actions)
    6. Assert: log entry has correct action (CREATE, UPDATE, or DELETE)
    7. Assert: log entry has entityId matching the resource
    8. For CREATE: assert newValue is defined and non-null
    9. For UPDATE: assert oldValue and newValue are both defined

Do NOT attempt to test every single endpoint — that would exceed context budget. Cover the core 10-12 mutations that are most audit-critical.
